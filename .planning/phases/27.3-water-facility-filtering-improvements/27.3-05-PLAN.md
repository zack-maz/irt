```typescript
const WATER_TYPE_LABELS: Record<WaterFacilityType, string> = {
  dam: 'Dam',
  reservoir: 'Reservoir',
  desalination: 'Desalination Plant',
};

const NEAR_UNKNOWN_RE = /\bnear\s+unknown\s*$/i;

export function getWaterFacilityDisplayName(facility: WaterFacility): string {
  const typeLabel = WATER_TYPE_LABELS[facility.facilityType];

  const raw = facility.label?.trim();
  if (raw && !NEAR_UNKNOWN_RE.test(raw)) {
    return raw; // <-- this returns the bare "Reservoir" when label === "Reservoir"
  }

  if (facility.linkedRiver) {
    return `${typeLabel} on ${facility.linkedRiver.name}`;
  }

  if (facility.nearestCity) {
    return `${typeLabel} near ${facility.nearestCity.name}`;
  }

  const latStr = `${Math.abs(facility.lat).toFixed(2)}°${facility.lat >= 0 ? 'N' : 'S'}`;
  const lngStr = `${Math.abs(facility.lng).toFixed(2)}°${facility.lng >= 0 ? 'E' : 'W'}`;
  return `${typeLabel} at ${latStr}, ${lngStr}`;
}
```

**After:**

```typescript
/**
 * Word-bounded, case-insensitive match for the token "dam" in a name — used to
 * reclassify misclassified-reservoir facilities whose OSM name is the ground truth
 * (e.g. "Hub Dam" tagged as natural=water/water=reservoir).
 * Matches: "Hub Dam", "Mosul Dam", "DAM" / "dam" anywhere with word boundaries.
 * Does NOT match: "Damascus", "Adamana" — word boundaries fail there.
 * Phase 27.3 Plan 05 / UAT Test 8b.
 */
const DAM_IN_NAME_RE = /\bdam\b/i;

export function classifyWaterType(tags: Record<string, string>): WaterFacilityType | null {
  // Tag-based classification (original logic preserved verbatim)
  let result: WaterFacilityType | null = null;
  if (tags['waterway'] === 'dam') result = 'dam';
  else if (tags['man_made'] === 'dam') result = 'dam';
  else if (tags['natural'] === 'water' && tags['water'] === 'reservoir') result = 'reservoir';
  else if (tags['landuse'] === 'reservoir') result = 'reservoir';
  else if (tags['man_made'] === 'desalination_plant') result = 'desalination';
  else if (tags['water_works'] === 'desalination') result = 'desalination';

  // Phase 27.3 Plan 05 / UAT Test 8b: name-based override for reservoir → dam.
  // An OSM element tagged as a reservoir but named "<Something> Dam" is almost
  // always a dam whose mapper only tagged the impounded water surface. The
  // override is one-directional: reservoir → dam only, never the reverse.
  if (result === 'reservoir') {
    const name = tags['name:en'] ?? tags['name'] ?? '';
    if (DAM_IN_NAME_RE.test(name)) {
      return 'dam';
    }
  }

  return result;
}
```
