**haversine pattern** (lines 106-115 — reuse as-is for river linkage distance):

```typescript
function haversine(lat1: number, lng1: number, lat2: number, lng2: number): number {
  const R = 6371;
  const toRad = (d: number) => (d * Math.PI) / 180;
  const dLat = toRad(lat2 - lat1);
  const dLng = toRad(lng2 - lng1);
  const a =
    Math.sin(dLat / 2) ** 2 +
    Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.sin(dLng / 2) ** 2;
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
}
```

**Enrichment extraction pattern** — follow `assignBasinStress` (basinLookup.ts lines 141-206) as the model for enrichment functions: load static data at module init, pure function takes `(lat, lng)`, returns optional typed object:

```typescript
// Load rivers.json at module init (mirrors basinLookup.ts pattern):
const rivers: RiverFeature[] = JSON.parse(readFileSync(riversPath, 'utf-8')).features;

// City data (42 entries) — inline or imported from a minimal JSON:
interface CityEntry {
  name: string;
  lat: number;
  lng: number;
  population: number;
}
const CITY_DATA: CityEntry[] = [
  /* 42 cities with population */
];

function findNearestCity(
  lat: number,
  lng: number,
): { name: string; distanceKm: number; population: number } | null {
  let nearest = null as { name: string; distanceKm: number; population: number } | null;
  for (const city of CITY_DATA) {
    const d = haversine(lat, lng, city.lat, city.lng);
    if (!nearest || d < nearest.distanceKm)
      nearest = { name: city.name, distanceKm: d, population: city.population };
  }
  return nearest && nearest.distanceKm <= 150 ? nearest : null;
}

function linkRiver(lat: number, lng: number): { name: string; distanceKm: number } | null {
  let nearest = null as { name: string; distanceKm: number } | null;
  for (const f of rivers) {
    const coords: number[][] =
      f.geometry.type === 'LineString'
        ? (f.geometry.coordinates as number[][])
        : (f.geometry.coordinates as number[][][]).flat();
    for (const [lng2, lat2] of coords) {
      const d = haversine(lat, lng, lat2!, lng2!);
      if (!nearest || d < nearest.distanceKm)
        nearest = { name: f.properties.name ?? '', distanceKm: d };
    }
  }
  return nearest && nearest.distanceKm <= 20 ? nearest : null;
}

function extractCapacityTags(
  tags: Record<string, string>,
): { height?: number; volume?: number; area?: number } | null {
  const height = tags['height'] ? parseFloat(tags['height']) : undefined;
  const volume = tags['volume'] ? parseFloat(tags['volume']) : undefined;
  const area = tags['area'] ? parseFloat(tags['area']) : undefined;
  if (height === undefined && volume === undefined && area === undefined) return null;
  return { height, volume, area };
}
```
