10. Create `extractCapacityTags` function (exported for testing):

    ```typescript
    export function extractCapacityTags(
      tags: Record<string, string>,
    ): { height?: number; volume?: number; area?: number } | null {
      const height = tags['height'] ? parseFloat(tags['height']) : undefined;
      const volume =
        (tags['volume'] ?? tags['capacity'])
          ? parseFloat(tags['volume'] ?? tags['capacity']!)
          : undefined;
      const area = tags['area'] ? parseFloat(tags['area']) : undefined;
      if (height === undefined && volume === undefined && area === undefined) return null;
      return {
        ...(height !== undefined && !isNaN(height) && { height }),
        ...(volume !== undefined && !isNaN(volume) && { volume }),
        ...(area !== undefined && !isNaN(area) && { area }),
      };
    }
    ```

11. Create `findNearestCity` function (exported for testing):

    ```typescript
    export function findNearestCity(
      lat: number,
      lng: number,
    ): { name: string; distanceKm: number; population: number } | null {
      let nearest: { name: string; distanceKm: number; population: number } | null = null;
      for (const city of CITY_DATA) {
        const d = haversine(lat, lng, city.lat, city.lng);
        if (!nearest || d < nearest.distanceKm) {
          nearest = { name: city.name, distanceKm: Math.round(d), population: city.population };
        }
      }
      return nearest && nearest.distanceKm <= 150 ? nearest : null;
    }
    ```

    The 150km threshold ensures only meaningfully proximate cities are linked.

12. Create `linkRiver` function (exported for testing):

    ```typescript
    export function linkRiver(
      lat: number,
      lng: number,
    ): { name: string; distanceKm: number } | null {
      let nearest: { name: string; distanceKm: number } | null = null;
      for (const feature of riversData.features) {
        const coords: number[][] =
          feature.geometry.type === 'LineString'
            ? (feature.geometry.coordinates as number[][])
            : (feature.geometry.coordinates as number[][][]).flat();
        for (const coord of coords) {
          const [lng2, lat2] = coord;
          if (lng2 === undefined || lat2 === undefined) continue;
          const d = haversine(lat, lng, lat2, lng2);
          if (!nearest || d < nearest.distanceKm) {
            nearest = { name: feature.properties.name, distanceKm: Math.round(d) };
          }
        }
      }
      return nearest && nearest.distanceKm <= 20 ? nearest : null;
    }
    ```

    The 20km threshold links facilities to rivers they likely depend on without false positives.
