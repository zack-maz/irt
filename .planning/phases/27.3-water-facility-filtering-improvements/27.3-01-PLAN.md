10. Add two new exported functions mirroring saveDevLLMCache/loadDevLLMCache exactly:

```typescript
/**
 * Save normalized water facilities + filter stats to local file. No-op in production.
 *
 * WHY: Dams/reservoirs do not change location. On dev server restart, Redis is cold
 * but Overpass is slow (60-90s for reservoir query). This file cache means the second
 * dev start is instant for base facility data. Precipitation is NOT cached here —
 * that stays real-time via the separate /api/water/precip endpoint.
 */
export function saveDevWaterCache<T>(data: T): void {
  if (!isDev) return;
  try {
    if (!existsSync(DEV_CACHE_DIR)) {
      mkdirSync(DEV_CACHE_DIR, { recursive: true });
    }
    const entry: DevCacheEntry<T> = { data, savedAt: Date.now() };
    writeFileSync(WATER_FACILITIES_FILE, JSON.stringify(entry));
    log.info('saved water facilities to dev file cache');
  } catch (err) {
    log.warn({ err }, 'failed to write water dev file cache');
  }
}

/**
 * Load water facilities from local file. Returns null in production, missing, or stale.
 */
export function loadDevWaterCache<T>(): T | null {
  if (!isDev) return null;
  try {
    if (!existsSync(WATER_FACILITIES_FILE)) return null;
    const raw = readFileSync(WATER_FACILITIES_FILE, 'utf-8');
    const entry = JSON.parse(raw) as DevCacheEntry<T>;
    const age = Date.now() - entry.savedAt;
    if (age > MAX_AGE_MS) {
      log.info({ ageMs: age }, 'water dev file cache too old, ignoring');
      return null;
    }
    log.info(
      { ageMs: age, ageMin: Math.round(age / 60_000) },
      'loaded water facilities from dev file cache',
    );
    return entry.data;
  } catch (err) {
    log.warn({ err }, 'failed to read water dev file cache');
    return null;
  }
}
```

### 2A.3 — Add river data load and per-river bbox pre-computation (user adjustment 3 + REVIEWS.md MEDIUM concern)

interface RiverFeature {
  type: 'Feature';
  properties: { name?: string; scalerank?: number; compositeHealth?: number };
  geometry: { type: 'MultiLineString' | 'LineString'; coordinates: number[][] | number[][][] };
}

interface RiverWithBbox {
  name: string;
  /** Flattened array of [lng, lat] pairs */
  coords: number[][];
  /** Bbox with 20km margin already added */
  minLat: number;
  maxLat: number;
  minLng: number;
  maxLng: number;
}

const riversPath = resolve(__dirname, '../../src/data/rivers.json');
const riversData: { features: RiverFeature[] } = JSON.parse(readFileSync(riversPath, 'utf-8'));

/** Degrees of latitude per 20km ≈ 20 / 111 = 0.180° */
const BBOX_MARGIN_LAT = 20 / 111;
/** Degrees of longitude per 20km varies with latitude; use 0.21° as a conservative margin at mid-ME lat ~32° */
const BBOX_MARGIN_LNG = 20 / (111 * Math.cos((32 * Math.PI) / 180));

/** Pre-compute bbox for each river so linkRiver can skip haversine for facilities outside the zone */
const RIVERS_WITH_BBOX: RiverWithBbox[] = riversData.features.map((f) => {
  const flat: number[][] =
    f.geometry.type === 'LineString'
      ? (f.geometry.coordinates as number[][])
      : (f.geometry.coordinates as number[][][]).flat();
  let minLat = Infinity;
  let maxLat = -Infinity;
  let minLng = Infinity;
  let maxLng = -Infinity;
  for (const [lng, lat] of flat) {
    if (lng === undefined || lat === undefined) continue;
    if (lat < minLat) minLat = lat;
    if (lat > maxLat) maxLat = lat;
    if (lng < minLng) minLng = lng;
    if (lng > maxLng) maxLng = lng;
  }
  return {
    name: f.properties.name ?? '',
    coords: flat,
    minLat: minLat - BBOX_MARGIN_LAT,
    maxLat: maxLat + BBOX_MARGIN_LAT,
    minLng: minLng - BBOX_MARGIN_LNG,
    maxLng: maxLng + BBOX_MARGIN_LNG,
  };
});

/** Strip unit suffixes like "85 m" or "1.2e6 m3" before parseFloat (REVIEWS.md LOW concern) */
function parseNumeric(value: string): number {
  // Keep digits, decimal point, minus sign, and scientific notation 'e'
  const cleaned = value.replace(/[^0-9.eE+-]/g, '').trim();
  if (cleaned === '') return NaN;
  const n = parseFloat(cleaned);
  return Number.isFinite(n) ? n : NaN;
}

export function extractCapacityTags(
  tags: Record<string, string>,
): { height?: number; volume?: number; area?: number } | null {
  const rawHeight = tags['height'];
  const rawVolume = tags['volume'] ?? tags['capacity'];
  const rawArea = tags['area'];
  const height = rawHeight ? parseNumeric(rawHeight) : NaN;
  const volume = rawVolume ? parseNumeric(rawVolume) : NaN;
  const area = rawArea ? parseNumeric(rawArea) : NaN;
  const out: { height?: number; volume?: number; area?: number } = {};
  if (Number.isFinite(height)) out.height = height;
  if (Number.isFinite(volume)) out.volume = volume;
  if (Number.isFinite(area)) out.area = area;
  return Object.keys(out).length > 0 ? out : null;
}

export function findNearestCity(
  lat: number,
  lng: number,
): { name: string; distanceKm: number; population: number } | null {
  let nearest: { name: string; distanceKm: number; population: number } | null = null;
  for (const city of CITY_DATA) {
    const d = haversine(lat, lng, city.lat, city.lng);
    if (!nearest || d < nearest.distanceKm) {
      nearest = { name: city.name, distanceKm: Math.round(d), population: city.population };
    }
  }
  return nearest && nearest.distanceKm <= 150 ? nearest : null;
}

/**
 * Link facility to nearest river with 20km threshold, using per-river bbox
 * pre-filter to skip haversine computation for facilities outside the river zone.
 * Addresses REVIEWS.md MEDIUM performance concern (user adjustment 3).
 */
export function linkRiver(lat: number, lng: number): { name: string; distanceKm: number } | null {
  let nearest: { name: string; distanceKm: number } | null = null;
  for (const river of RIVERS_WITH_BBOX) {
    // Bbox guard — skips haversine for ~80% of facilities (user adjustment 3)
    if (lat < river.minLat || lat > river.maxLat) continue;
    if (lng < river.minLng || lng > river.maxLng) continue;
    // Inside bbox — iterate vertices for exact distance
    for (const coord of river.coords) {
      const lng2 = coord[0];
      const lat2 = coord[1];
      if (lng2 === undefined || lat2 === undefined) continue;
      const d = haversine(lat, lng, lat2, lng2);
      if (!nearest || d < nearest.distanceKm) {
        nearest = { name: river.name, distanceKm: Math.round(d) };
      }
    }
  }
  return nearest && nearest.distanceKm <= 20 ? nearest : null;
}

/**
 * Holistic reservoir notability per user adjustments 1+2 and REVIEWS.md MEDIUM concern.
 *
 * Reservoir passes if ANY of:
 *   - has wikidata ref
 *   - has wikipedia ref (any language)
 *   - has a name AND is in a priority country (conflict-relevant by geography)
 *
 * This is NOT strict wikidata-only — that was shown to overcorrect. The hybrid
 * fallback keeps strategically relevant reservoirs in Iran/Iraq/Syria/Lebanon/
 * Israel/Jordan/Afghanistan even when their wikidata coverage is sparse, while
 * still dropping the 5000+ miniscule unnamed reservoirs everywhere else.
 */
export function isReservoirNotable(
  tags: Record<string, string>,
  lat: number,
  lng: number,
): boolean {
  if (tags.wikidata) return true;
  if (tags.wikipedia) return true;
  if (Object.keys(tags).some((k) => k.startsWith('wikipedia:'))) return true;
  // Hybrid fallback: named reservoir in a priority (conflict-zone) country
  if (hasName(tags) && isPriorityCountry(lat, lng)) return true;
  return false;
}

export function normalizeWaterElement(
  el: OverpassElement,
  stressLookup: (lat: number, lng: number) => WaterStressIndicators,
  rejections?: WaterFilterStats['rejections'],
): WaterFacility | null {
  if (!el.tags) return null;
  const facilityType = classifyWaterType(el.tags);
  if (!facilityType) return null;

  const lat = el.lat ?? el.center?.lat;
  const lon = el.lon ?? el.center?.lon;
  if (lat === undefined || lon === undefined) return null;

  // Geographic exclusion
  if (isExcludedLocation(lat, lon)) {
    if (rejections) rejections.excluded_location++;
    return null;
  }

  // Reservoir filter: holistic (user adjustments 1+2) — applies to ALL countries,
  // but priority countries get the named-fallback.
  if (facilityType === 'reservoir' && !isReservoirNotable(el.tags, lat, lon)) {
    if (rejections) rejections.not_notable++;
    return null;
  }

  // Dams: require a name in non-priority countries (priority keeps relaxed).
  if (facilityType === 'dam' && !isPriorityCountry(lat, lon) && !hasName(el.tags)) {
    if (rejections) rejections.no_name++;
    return null;
  }

  // desalination: always passes (very low volume, all strategic)

  // Enrichment (compute once, spread if present)
  const capacity = extractCapacityTags(el.tags);
  const nearestCity = findNearestCity(lat, lon);
  const linkedRiver = linkRiver(lat, lon);

  return {
    id: `water-${el.id}`,
    type: 'water',
    facilityType,
    lat,
    lng: lon,
    label: extractLabel(el.tags, facilityType),
    operator:
      el.tags.operator && isLatin(el.tags.operator) ? toTitleCase(el.tags.operator) : undefined,
    osmId: el.id,
    stress: stressLookup(lat, lon),
    ...(capacity && { capacity }),
    ...(nearestCity && { nearestCity }),
    ...(linkedRiver && { linkedRiver }),
  };
}

async function fetchFacilityType(
  entry: { label: string; nwr: string },
  stats: WaterFilterStats,
): Promise<WaterFacility[]> {
  const query = buildQuery(entry.nwr);
  for (const url of [OVERPASS_URL, OVERPASS_FALLBACK]) {
    try {
      const res = await fetch(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: `data=${encodeURIComponent(query)}`,
        signal: AbortSignal.timeout(TIMEOUT_MS),
      });
      if (!res.ok) {
        log.warn(
          { facilityType: entry.label, url, status: res.status },
          'Overpass returned error status',
        );
        continue;
      }
      const json = (await res.json()) as { elements: OverpassElement[] };

      // Track raw Overpass count (before any filtering)
      stats.rawCounts[entry.label] = (stats.rawCounts[entry.label] ?? 0) + json.elements.length;

      const facilities: WaterFacility[] = [];
      for (const el of json.elements) {
        const facility = normalizeWaterElement(el, assignBasinStress, stats.rejections);
        if (facility) facilities.push(facility);
      }

      // Track post-normalization, pre-dedup count
      stats.filteredCounts[entry.label] =
        (stats.filteredCounts[entry.label] ?? 0) + facilities.length;

      log.info(
        { facilityType: entry.label, raw: json.elements.length, kept: facilities.length },
        'fetched facilities',
      );
      return facilities;
    } catch (err) {
      log.warn({ err, facilityType: entry.label, url }, 'Overpass request failed');
    }
  }
  log.warn({ facilityType: entry.label }, 'all URLs failed, skipping');
  return [];
}

### 2A.11 — Rewrite fetchWaterFacilities with stats, dedup, and dev file cache fallback

```typescript
export async function fetchWaterFacilities(): Promise<{
  facilities: WaterFacility[];
  stats: WaterFilterStats;
}> {
  const stats: WaterFilterStats = {
    rawCounts: {},
    filteredCounts: {},
    rejections: { excluded_location: 0, not_notable: 0, no_name: 0, duplicate: 0 },
    enriched: { withCapacity: 0, withCity: 0, withRiver: 0 },
  };

  const all: WaterFacility[] = [];
  let succeeded = 0;

  for (const entry of FACILITY_QUERIES) {
    try {
      const facilities = await fetchFacilityType(entry, stats);
      if (facilities.length > 0) {
        succeeded++;
        all.push(...facilities);
      }
    } catch (err) {
      log.warn({ facilityType: entry.label, err }, 'query failed, continuing');
    }
  }

  if (succeeded === 0) {
    throw new Error('All Overpass API instances failed for water facilities');
  }

  // Dedup by OSM id first (trivial cases)
  const uniqueById = new Map<string, WaterFacility>();
  for (const f of all) uniqueById.set(f.id, f);

  // Spatial dedup: node+way+relation for same physical feature within 50m
  const deduped: WaterFacility[] = [];
  const before = uniqueById.size;
  for (const f of uniqueById.values()) {
    const isDupe = deduped.some(
      (existing) =>
        existing.facilityType === f.facilityType &&
        haversine(existing.lat, existing.lng, f.lat, f.lng) < 0.05, // 50m
    );
    if (!isDupe) deduped.push(f);
  }
  stats.rejections.duplicate = before - deduped.length;

  // Enrichment coverage stats (REVIEWS.md suggestion)
  for (const f of deduped) {
    if (f.capacity) stats.enriched.withCapacity++;
    if (f.nearestCity) stats.enriched.withCity++;
    if (f.linkedRiver) stats.enriched.withRiver++;
  }

  log.info(
    { total: deduped.size ?? deduped.length, succeeded, stats },
    'water facilities fetch complete',
  );
  return { facilities: deduped, stats };
}
```
