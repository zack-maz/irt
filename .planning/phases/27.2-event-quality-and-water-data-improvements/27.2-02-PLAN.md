```typescript
// Build news context for LLM prompt enrichment
let newsContext: NewsContext | undefined;
try {
  const newsForLLM = await cacheGetSafe<NewsCluster[]>('news:gdelt', 0);
  if (newsForLLM?.data) {
    newsContext = new Map();
    for (const cluster of newsForLLM.data) {
      for (const article of cluster.articles) {
        const domain = article.domain ?? extractDomain(article.url);
        const tier = getSourceTier(article.source, domain);
        if (tier !== null && tier <= 2) {
          // Only Tier 1+2 for LLM context (per D-02: "high-tier sources")
          newsContext.set(article.url, {
            title: article.title,
            snippet: article.summary,
            tier,
          });
        }
      }
    }
    log.info({ contextSize: newsContext.size }, 'LLM: built news context for prompt enrichment');
  }
} catch {
  log.warn('LLM: failed to build news context (non-fatal)');
}
```
