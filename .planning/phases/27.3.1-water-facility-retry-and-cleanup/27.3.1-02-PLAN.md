    function extractLabel(tags: Record<string, string>, facilityType: WaterFacilityType): string {
      const en = tags['name:en'];
      if (en && en.trim() && isLatin(en)) return toTitleCase(en);
      const raw = tags['name'] || '';
      if (raw && isLatin(raw)) return toTitleCase(raw);
