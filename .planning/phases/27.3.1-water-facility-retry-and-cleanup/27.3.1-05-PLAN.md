    **Step 1 — Create server/lib/waterSnapshot.ts.**

    ```typescript
    import { readFileSync, existsSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import type { WaterFacility } from '../types.js';
    import type { WaterFilterStats } from '../adapters/overpass-water.js';
    import { logger } from './logger.js';

    const log = logger.child({ module: 'water-snapshot' });

    const __dirname = dirname(fileURLToPath(import.meta.url));
    const SNAPSHOT_PATH = resolve(__dirname, '../../src/data/water-facilities.json');

    /**
     * Phase 27.3.1 R-04 — committed JSON snapshot of pre-enriched water
     * facilities. The snapshot sits at the bottom of the runtime resolution tier:
     *   Redis → dev file cache → snapshot → Overpass
     *
     * This is the production cold-start floor. Refresh is manual (npm run
     * refresh:water) — the script regenerates the file, the developer reviews
     * the diff, the commit ships. Overpass is NEVER reached from a user-facing
     * request: on a cold Redis + missing dev cache, the snapshot serves the
     * first miss and populates Redis, taking Overpass off the synchronous
     * request path entirely (R-07 multi-user-resilience invariant).
     */
    export interface WaterSnapshot {
      generatedAt: string;
      facilities: WaterFacility[];
      stats: WaterFilterStats;
    }

    let cachedSnapshot: WaterSnapshot | null | undefined = undefined;

    /**
     * Load the committed water facility snapshot. Returns null when the file
     * does not exist or fails structural validation. Cached in-module so
     * subsequent calls are O(1) — the snapshot is immutable at runtime.
     */
    export function loadWaterSnapshot(): WaterSnapshot | null {
      if (cachedSnapshot !== undefined) return cachedSnapshot;

      try {
        if (!existsSync(SNAPSHOT_PATH)) {
          log.info({ path: SNAPSHOT_PATH }, 'water snapshot file absent; cold-start will hit Overpass');
          cachedSnapshot = null;
          return null;
        }

        const raw = readFileSync(SNAPSHOT_PATH, 'utf-8');
        const parsed = JSON.parse(raw) as unknown;

        if (!isValidSnapshot(parsed)) {
          log.warn({ path: SNAPSHOT_PATH }, 'water snapshot failed structural validation; ignoring');
          cachedSnapshot = null;
          return null;
        }

        // R-08 D-30 — snapshot source overrides any persisted source value
        const snapshot: WaterSnapshot = {
          ...parsed,
          stats: { ...parsed.stats, source: 'snapshot', generatedAt: parsed.generatedAt },
        };

        log.info(
          { count: snapshot.facilities.length, generatedAt: snapshot.generatedAt },
          'loaded water snapshot',
        );
        cachedSnapshot = snapshot;
        return snapshot;
      } catch (err) {
        log.warn({ err, path: SNAPSHOT_PATH }, 'failed to load water snapshot');
        cachedSnapshot = null;
        return null;
      }
    }

    /** Reset the in-module cache. Test-only. */
    export function __resetSnapshotCacheForTests(): void {
      cachedSnapshot = undefined;
    }

    function isValidSnapshot(v: unknown): v is WaterSnapshot {
      if (!v || typeof v !== 'object') return false;
      const o = v as Record<string, unknown>;
      if (typeof o.generatedAt !== 'string') return false;
      if (!Array.isArray(o.facilities)) return false;
      if (!o.stats || typeof o.stats !== 'object') return false;
      return true;
    }
    ```

    **Step 2 — Create scripts/refresh-water-facilities.ts.**

    ```typescript
    #!/usr/bin/env node
    /**
     * Phase 27.3.1 R-04 — manual refresh script for the committed water
     * facilities snapshot (src/data/water-facilities.json).
     *
     * Runs the full server-side fetch pipeline (Overpass → classify → D-05+D-06
     * gate → enrich → basin stress → reverse-geocode unnamed) and writes the
     * result as a committed JSON file.
     *
     * Usage:  npm run refresh:water
     *
     * Intended cadence: run when you want fresh data. Commit the diff for
     * review before merging. Do NOT run as part of CI — Overpass rate limits
     * and reverse-geocode's 1 req/s pace make a fresh refresh take several
     * minutes.
     *
     * Model: mirrors scripts/extract-rivers.ts / scripts/extract-aqueduct-basins.ts.
     */

    import { writeFileSync, mkdirSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import { fetchWaterFacilities } from '../server/adapters/overpass-water.js';
    import { labelUnnamedFacilities } from '../server/lib/waterLabeling.js';

    const __dirname = dirname(fileURLToPath(import.meta.url));
    const OUT_PATH = resolve(__dirname, '../src/data/water-facilities.json');

    async function main(): Promise<void> {
      console.log('=== Refreshing water facility snapshot ===');
      console.log(`Output: ${OUT_PATH}`);
      console.log();

      const start = Date.now();

      console.log('[1/3] Fetching from Overpass (3 queries: dams, reservoirs, desalination)...');
      const { facilities: raw, stats } = await fetchWaterFacilities();
      console.log(`  → ${raw.length} facilities admitted (Overpass fetch: ${Date.now() - start}ms)`);
      for (const [type, count] of Object.entries(stats.filteredCounts)) {
        const rawCount = stats.rawCounts[type] ?? 0;
        console.log(`    ${type}: ${count} kept / ${rawCount} raw`);
      }

      console.log();
      console.log('[2/3] Reverse-geocoding unnamed facilities (~1 req/s rate limit)...');
      const labeled = await labelUnnamedFacilities(raw);
      const unnamed = labeled.filter((f) => f.label.startsWith('Dam ') || f.label.startsWith('Reservoir ') || f.label.startsWith('Desalination Plant ')).length;
      console.log(`  → ${unnamed} facilities have "Type near <City>" reverse-geocoded labels`);

      console.log();
      console.log('[3/3] Writing snapshot...');
      const snapshot = {
        generatedAt: new Date().toISOString(),
        facilities: labeled,
        stats: { ...stats, source: 'snapshot', generatedAt: new Date().toISOString() },
      };

      // Ensure output directory exists (repo normally has src/data; defensive)
      mkdirSync(resolve(__dirname, '../src/data'), { recursive: true });

      // Pretty-print (2-space) so PR diffs are reviewable. File size grows ~30% vs
      // minified but remains well under 1MB for ~800 facilities.
      writeFileSync(OUT_PATH, JSON.stringify(snapshot, null, 2));

      const elapsed = ((Date.now() - start) / 1000).toFixed(1);
      console.log(`  → ${labeled.length} facilities written (${elapsed}s total)`);
      console.log();
      console.log(`✓ Snapshot written to ${OUT_PATH}`);
      console.log();
      console.log('Next steps:');
      console.log('  1. Review the diff: git diff src/data/water-facilities.json | head -50');
      console.log('  2. Spot-check facility counts and sample labels');
      console.log('  3. Commit: git add src/data/water-facilities.json && git commit -m "data(27.3.1): refresh water facilities snapshot"');
    }

    main().catch((err) => {
      console.error(err);
      process.exit(1);
    });
    ```
