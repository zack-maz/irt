    ```typescript
    /**
     * Per-facility admission decision. Returns a structured verdict so callers
     * can both tally the rejection bucket AND know why. Replaces the scattered
     * rejection branches in `normalizeWaterElement` (Phase 27.3.1 R-06 D-20).
     *
     * Decision order (short-circuit on first reject):
     *   1. excluded_location  — isExcludedLocation(coords)
     *   2. no_name            — !hasName(tags)                        [R-03 D-05]
     *   3. not_notable        — !(isNotable || inPriority || hasCapacityData) [R-03 D-06]
     *   4. low_score          — score < MIN_NOTABILITY_SCORE          [R-03 D-08]
     *   5. no_city            — reservoir-specific (R-03 D-09)
     *   6. admit
     *
     * The duplicate bucket is decided post-normalization in fetchWaterFacilities
     * (spatial dedup), not here.
     */
    export type AdmissionVerdict =
      | { verdict: 'admit' }
      | { verdict: 'reject'; bucket: keyof WaterFilterStats['rejections'] };

    export function computeAdmissionDecision(
      tags: Record<string, string>,
      lat: number,
      lng: number,
      facilityType: WaterFacilityType,
      inPriority: boolean,
      score: number,
      nearestCity: ReturnType<typeof findNearestCity>,
    ): AdmissionVerdict {
      if (isExcludedLocation(lat, lng)) return { verdict: 'reject', bucket: 'excluded_location' };

      // R-03 D-05: name is mandatory
      if (!hasName(tags)) return { verdict: 'reject', bucket: 'no_name' };

      // R-03 D-06: compound notability signal
      const passesCompound = isNotable(tags) || inPriority || hasCapacityData(tags);
      if (!passesCompound) return { verdict: 'reject', bucket: 'not_notable' };

      // R-03 D-08: secondary score floor
      if (score < MIN_NOTABILITY_SCORE) return { verdict: 'reject', bucket: 'low_score' };

      // R-03 D-09: reservoir no_city rule (preserved from Phase 27.3 Plan 05)
      const hasWikiRef =
        !!tags.wikidata ||
        !!tags.wikipedia ||
        Object.keys(tags).some((k) => k.startsWith('wikipedia:'));
      const isNamedInPriorityCountry = inPriority && hasName(tags);
      if (facilityType === 'reservoir' && !nearestCity && !hasWikiRef && !isNamedInPriorityCountry) {
        return { verdict: 'reject', bucket: 'no_city' };
      }

      return { verdict: 'admit' };
    }
    ```

    ```typescript
    export function normalizeWaterElement(
      el: OverpassElement,
      stressLookup: (lat: number, lng: number) => WaterStressIndicators,
      rejections?: WaterFilterStats['rejections'],
      byTypeBucket?: WaterFilterStats['byTypeRejections'][string],
    ): WaterFacility | null {
      if (!el.tags) return null;
      const facilityType = classifyWaterType(el.tags);
      if (!facilityType) return null;

      const lat = el.lat ?? el.center?.lat;
      const lon = el.lon ?? el.center?.lon;
      if (lat === undefined || lon === undefined) return null;

      const inPriority = isPriorityCountry(lat, lon);
      const score = computeNotabilityScore(el.tags, facilityType, inPriority);
      const nearestCity = findNearestCity(lat, lon);

      const decision = computeAdmissionDecision(el.tags, lat, lon, facilityType, inPriority, score, nearestCity);
      if (decision.verdict === 'reject') {
        if (rejections) rejections[decision.bucket]++;
        if (byTypeBucket) byTypeBucket[decision.bucket]++;
        return null;
      }

      const capacity = extractCapacityTags(el.tags);
      const linkedRiver = linkRiver(lat, lon);

      return {
        id: `water-${el.id}`,
        type: 'water',
        facilityType,
        lat,
        lng: lon,
        label: extractLabel(el.tags, facilityType),
        operator:
          el.tags.operator && isLatin(el.tags.operator) ? toTitleCase(el.tags.operator) : undefined,
        osmId: el.id,
        stress: stressLookup(lat, lon),
        notabilityScore: score,
        ...(capacity && { capacity }),
        ...(nearestCity && { nearestCity }),
        ...(linkedRiver && { linkedRiver }),
      };
    }
    ```

    Reads top-to-bottom: classify → coords → score → admission decision → (reject | build).
